-- Scan All Remote Events/Functions
function ScanRemotes()
    RemoteData.RemoteEvents = {}
    RemoteData.RemoteFunctions = {}
    
    local function ScanInstance(instance, path)
        for _, child in pairs(instance:GetChildren()) do
            pcall(function()
                local childPath = path .. "/" .. child.Name
                
                if child:IsA("RemoteEvent") then
                    table.insert(RemoteData.RemoteEvents, {
                        Name = child.Name,
                        Path = childPath,
                        Instance = child,
                        Parent = instance.Name
                    })
                elseif child:IsA("RemoteFunction") then
                    table.insert(RemoteData.RemoteFunctions, {
                        Name = child.Name,
                        Path = childPath,
                        Instance = child,
                        Parent = instance.Name
                    })
                end
                
                ScanInstance(child, childPath)
            end)
        end
    end
    
    local locations = {
        game:GetService("Workspace"),
        game:GetService("ReplicatedStorage"),
        game:GetService("ReplicatedFirst"),
        LocalPlayer:FindFirstChild("PlayerGui"),
        LocalPlayer:FindFirstChild("Backpack")
    }
    
    if LocalPlayer.Character then
        table.insert(locations, LocalPlayer.Character)
    end
    
    for _, location in pairs(locations) do
        if location then
            pcall(function()
                ScanInstance(location, location.Name)
            end)
        end
    end
    
    return #RemoteData.RemoteEvents + #RemoteData.RemoteFunctions
end

-- Test Remote Vulnerability
function TestRemote(remote, testType)
    local results = {
        Success = false,
        Response = nil,
        Error = nil,
        Vulnerable = false
    }
    
    pcall(function()
        if testType == "FireEmpty" and remote.Instance:IsA("RemoteEvent") then
            remote.Instance:FireServer()
            results.Success = true
            results.Response = "Fired without parameters"
            results.Vulnerable = true
            
        elseif testType == "FireValues" and remote.Instance:IsA("RemoteEvent") then
            -- Test common exploitable values
            remote.Instance:FireServer(999999)
            remote.Instance:FireServer("Money", 999999)
            remote.Instance:FireServer({Money = 999999, Level = 999})
            results.Success = true
            results.Response = "Fired with exploit values"
            results.Vulnerable = true
            
        elseif testType == "FireNegative" and remote.Instance:IsA("RemoteEvent") then
            remote.Instance:FireServer(-999999)
            results.Success = true
            results.Response = "Fired with negative values"
            results.Vulnerable = true
            
        elseif testType == "FireString" and remote.Instance:IsA("RemoteEvent") then
            remote.Instance:FireServer("EXPLOIT_TEST")
            remote.Instance:FireServer(string.rep("A", 10000))
            results.Success = true
            results.Response = "Fired with string payloads"
            results.Vulnerable = true
            
        elseif testType == "InvokeTest" and remote.Instance:IsA("RemoteFunction") then
            local response = remote.Instance:InvokeServer()
            results.Success = true
            results.Response = tostring(response)
            results.Vulnerable = true
        end
    end)
    
    if results.Vulnerable then
        table.insert(RemoteData.VulnerableRemotes, {
            Remote = remote,
            TestType = testType,
            Results = results,
            Time = os.date("%H:%M:%S")
        })
    end
    
    table.insert(RemoteData.TestResults, {
        Remote = remote.Name,
        TestType = testType,
        Results = results
    })
    
    return results
end

-- Fuzzer - Test multiple parameters
function FuzzRemote(remote, iterations)
    local testPayloads = {
        999999, -999999, 0, 1, -1,
        "Money", "Cash", "Gold", "Gems", "Admin",
        true, false, nil,
        {Money = 999999}, {Level = 999}, {Admin = true},
        string.rep("A", 1000),
        workspace, game, LocalPlayer
    }
    
    local successCount = 0
    
    for i = 1, iterations do
        for _, payload in pairs(testPayloads) do
            pcall(function()
                if remote.Instance:IsA("RemoteEvent") then
                    remote.Instance:FireServer(payload)
                    successCount = successCount + 1
                elseif remote.Instance:IsA("RemoteFunction") then
                    remote.Instance:InvokeServer(payload)
                    successCount = successCount + 1
                end
            end)
            task.wait(0.01)
        end
    end
    
    return successCount
end
